--> Running the server 
open 2 cmd and direct it to the folders
# Terminal 1 — Start FastAPI
cd model
pip install fastapi uvicorn joblib opencv-python 
uvicorn API:app --host 0.0.0.0 --port 8000

# Terminal 2 — Start Node.js
cd server
npm install express axios 
node server.js

# Terminal 3 — Ngrok
cd server
npm install -g ngrok

(Sign up in ngrok and get your token)
ngrok config add-authtoken YOUR_TOKEN_HERE
ngrok http 4000
